unsigned short  str_longitud (char* str)
{
    unsigned short j=0;

    while(str[j]>0){
        j++;
    }

    return j;
}

char mayuschar(char CARACTER){

    if( (CARACTER>=97) && (CARACTER<=122) )
        CARACTER-=32;

    return CARACTER;
}
