#include <stdio.h>
int factorial(int);

int main(void)
{
    int dato;

    printf("Ingrese Nro a factoriar:\t");
    scanf("%d", &dato);

    printf("\nEl factorial de %d es: %d",dato, factorial(dato));

    printf("\n\n");
    return 0;
}

int factorial(int x)
{
    int i, acum=1;
    for(i=1; i<=x; i++)
    {
        acum = acum * i;

    }
return acum;
}
